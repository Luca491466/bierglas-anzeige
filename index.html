  <script>

    const fillElement = document.getElementById('beerFill');

    const input = document.getElementById('chatterInput');


    function updateBeerGlass(chatterCount) {

      const minChatter = 2;

      const maxChatter = 100;


      // Begrenzung der Werte

      chatterCount = Math.max(minChatter, Math.min(maxChatter, chatterCount));


      // Füllstand berechnen in Prozent

      const percent = ((chatterCount - minChatter) / (maxChatter - minChatter)) * 100;

      fillElement.style.height = percent + "%";

    }


    // Initial

    updateBeerGlass(parseInt(input.value));


    // Simulations-Event

    input.addEventListener('input', () => {

      const value = parseInt(input.value);

      updateBeerGlass(value);

    });

  </script>
